import sys

ages = [1, 2, 3, 4]
it = iter(ages)
while True:
  try:
    print(next(it))
  except StopIteration:
    sys.exit()


def fibonacci(n):  # 包含yield的函数，生成器：斐波那契
  a, b, count = 0, 1, 0
  while True:
    if (count > n):
      return
    yield a
    a, b = b, a+b
    count += 1

it = fibonacci(10)  # 生成器返回一个迭代器

while True:
  try:
    print(next(it), end=" ")
  except StopIteration:
    sys.exit()
