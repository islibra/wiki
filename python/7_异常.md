```python
import sys

try:
  f = open('/tmp/status.json', 'r')
  s = f.readline()
# i = int(s.strip())
except OSError as err:
  print('OS Error: {0}'.format(err))
except (NameError, ValueError):  # 处理多个异常
  print('Could not convert data to integer.')
except:  # 通配所有异常
  print('Unexcepted error:', sys.exx_info()[0])
  raise
else:  # 无任何异常时执行
  print(s)
  f.close()
```
